    (function setAmountFromQuery(){const el = $('#amount'); const params = new URLSearchParams(location.search); const raw = params.get('amt'); const num = raw && raw.match(/^\d{1,7}$/) ? Number(raw) : null; const amt = num ?? Number(el.dataset.defaultAmount || 815); el.textContent = formatUSD(amt);})();

    (function confetti(){

      const c = document.getElementById('confetti');

      const ctx = c.getContext('2d');

      let W, H, pieces = [], running = false, raf;

      function resize(){W = c.width = innerWidth; H = c.height = innerHeight;}

      window.addEventListener('resize', resize, {passive:true}); resize();

      function spawn(n=180){pieces = Array.from({length:n}, () => ({x: Math.random()*W, y: -20 - Math.random()*H*0.5, r: 4 + Math.random()*6, s: 1 + Math.random()*2, a: Math.random()*Math.PI*2, hue: [46, 30, 160][Math.floor(Math.random()*3)],}));}

      function draw(){ctx.clearRect(0,0,W,H); for(const p of pieces){p.y += p.s * 2; p.x += Math.sin((p.y+p.a)/30)*1.2; p.a += 0.03; ctx.save(); ctx.translate(p.x, p.y); ctx.rotate(p.a); ctx.fillStyle = `hsl(${p.hue} 90% 60%)`; ctx.fillRect(-p.r/2, -p.r/2, p.r, p.r*0.6); ctx.restore();} pieces = pieces.filter(p => p.y < H + 20); if (running) raf = requestAnimationFrame(draw);}      

      function startBurst(durationMs=3500){running = true; spawn(); draw(); if (navigator.vibrate) {try{navigator.vibrate([10, 40, 10]);}catch{}} setTimeout(()=>{running = false; cancelAnimationFrame(raf); ctx.clearRect(0,0,W,H);}, durationMs);}      

      startBurst();

      $('#celebrateBtn').addEventListener('click', ()=> startBurst(2800));

    })();
